```cpp
#include <cstdlib>
#include <cuda_runtime.h>

class TensorBuffer {
private:
    void* data_ptr;
    size_t size;
    bool is_device;  // 是否在 GPU 内存中

public:
    TensorBuffer(size_t size, bool device = false)
        : data_ptr(nullptr), size(size), is_device(device) {
        if (is_device) {
            cudaMalloc(&data_ptr, size);
        } else {
            data_ptr = std::malloc(size);
        }
    }

    ~TensorBuffer() {
        if (is_device) {
            cudaFree(data_ptr);
        } else {
            std::free(data_ptr);
        }
    }

    void copyFrom(const TensorBuffer& other) {
        cudaMemcpyKind kind = other.is_device
            ? (is_device ? cudaMemcpyDeviceToDevice : cudaMemcpyDeviceToHost)
            : (is_device ? cudaMemcpyHostToDevice : cudaMemcpyHostToHost);

        cudaMemcpy(data_ptr, other.data_ptr, size, kind);
    }
};
```

```cpp
// 假设在主机端已将卷积核拷贝至常量内存
#define TILE_SIZE 16
#define KERNEL_SIZE 3
#define RADIUS (KERNEL_SIZE / 2)

__constant__ float c_kernel[KERNEL_SIZE * KERNEL_SIZE];

__global__ void convolution_optimized(
    const float* input,
    float* output,
    int input_width,
    int input_height,
    int kernel_size
) {
    // 1. 分配共享内存，包含卷积核需要的 Halo / Border 区域
    extern __shared__ float shared_input[];

    int tx = threadIdx.x;pp
    int ty = threadIdx.y;
    int bx = blockIdx.x;
    int by = blockIdx.y;

    int shared_width = TILE_SIZE + KERNEL_SIZE - 1;

    // 计算当前线程负责的全局输出坐标
    int output_x = bx * TILE_SIZE + tx;
    int output_y = by * TILE_SIZE + ty;

    // 2. 协作将输入数据加载到共享内存
    for (int shared_y = ty; shared_y < shared_width; shared_y += blockDim.y) {
        for (int shared_x = tx; shared_x < shared_width; shared_x += blockDim.x) {
            int global_x = bx * TILE_SIZE + shared_x - RADIUS;
            int global_y = by * TILE_SIZE + shared_y - RADIUS;
            int shared_idx = shared_y * shared_width + shared_x;

            // 边界检查并加载数据，越界部分补零
            if (global_x >= 0 && global_x < input_width &&
                global_y >= 0 && global_y < input_height) {
                shared_input[shared_idx] = input[global_y * input_width + global_x];
            } else {
                shared_input[shared_idx] = 0.0f;
            }
        }
    }

    __syncthreads();  // 确保所有数据加载完毕

    // 3. 执行卷积计算
    if (output_x < input_width && output_y < input_height) {
        float sum = 0.0f;

        for (int ky = 0; ky < kernel_size; ky++) {
            for (int kx = 0; kx < kernel_size; kx++) {
                int shared_idx = (ty + ky) * shared_width + (tx + kx);
                int kernel_idx = ky * kernel_size + kx;
                sum += shared_input[shared_idx] * c_kernel[kernel_idx];
            }
        }

        output[output_y * input_width + output_x] = sum;
    }
}
```

```python
# 入门项目：图像分类推理服务
import torch
import torchvision.transforms as transforms
from PIL import Image


class ImageClassifier:
    def __init__(self, model_path):
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model = torch.load(model_path)
        self.model.to(self.device)
        self.model.eval()

        # 预处理
        self.transform = transforms.Compose([
            transforms.Resize(256),
            transforms.CenterCrop(224),
            transforms.ToTensor(),
            transforms.Normalize(
                mean=[0.485, 0.456, 0.406],
                std=[0.229, 0.224, 0.225],
            ),
        ])

    def predict(self, image_path):
        image = Image.open(image_path)
        input_tensor = self.transform(image).unsqueeze(0).to(self.device)

        with torch.no_grad():
            outputs = self.model(input_tensor)
            _, predicted = torch.max(outputs, 1)

        return predicted.item()

    def batch_predict(self, image_paths):
        images = [Image.open(path) for path in image_paths]
        batch_tensor = torch.stack([
            self.transform(img) for img in images
        ]).to(self.device)

        with torch.no_grad():
            outputs = self.model(batch_tensor)
            _, predicted = torch.max(outputs, 1)

        return predicted.cpu().numpy()
```

```python
import numpy as np
import pycuda.autoinit  # noqa: F401
import pycuda.driver as cuda
import tensorrt as trt


class TensorRTInference:
    def __init__(self, engine_path):
        self.logger = trt.Logger(trt.Logger.WARNING)
        self.engine = self.load_engine(engine_path)
        self.context = self.engine.create_execution_context()
        self.inputs, self.outputs, self.bindings = self.allocate_buffers()

    def load_engine(self, engine_path):
        with open(engine_path, "rb") as f:
            runtime = trt.Runtime(self.logger)
            return runtime.deserialize_cuda_engine(f.read())

    def allocate_buffers(self):
        inputs = []
        outputs = []
        bindings = []

        for binding in self.engine:
            size = (
                trt.volume(self.engine.get_binding_shape(binding))
                * self.engine.max_batch_size
            )
            dtype = trt.nptype(self.engine.get_binding_dtype(binding))

            # 分配 CPU 和 GPU 内存
            host_mem = cuda.pagelocked_empty(size, dtype)
            device_mem = cuda.mem_alloc(host_mem.nbytes)
            bindings.append(int(device_mem))

            buffer = {"host": host_mem, "device": device_mem}
            if self.engine.binding_is_input(binding):
                inputs.append(buffer)
            else:
                outputs.append(buffer)

        return inputs, outputs, bindings

    def infer(self, input_data):
        # 复制输入数据
        np.copyto(self.inputs[0]["host"], input_data.ravel())

        # 数据传输到 GPU
        for inp in self.inputs:
            cuda.memcpy_htod(inp["device"], inp["host"])

        # 执行推理
        self.context.execute_v2(bindings=self.bindings)

        # 数据传回 CPU
        for out in self.outputs:
            cuda.memcpy_dtoh(out["host"], out["device"])

        return [out["host"] for out in self.outputs]
```

```python
import torch
import torch.distributed as dist
import torch.multiprocessing as mp


class DistributedInference:
    def __init__(self, rank, world_size):
        self.rank = rank
        self.world_size = world_size
        self.setup_distributed()
        self.model = self.load_model_partition()

    def setup_distributed(self):
        # 初始化分布式环境
        dist.init_process_group(
            backend="nccl",
            init_method="tcp://localhost:29500",
            world_size=self.world_size,
            rank=self.rank,
        )

    def load_model_partition(self):
        # 加载模型的部分参数或分片，用于模型并行
        model = torch.load(f"model_partition_{self.rank}.pth")
        model = model.to(f"cuda:{self.rank}")
        model.eval()
        return model

    def pipeline_inference(self, input_data):
        # 流水线并行推理
        stages = 4
        chunk_size = len(input_data) // stages

        for stage in range(stages):
            # 处理当前阶段的数据
            stage_start = stage * chunk_size
            stage_end = (stage + 1) * chunk_size
            stage_data = input_data[stage_start:stage_end]

            # 在当前 GPU 上处理
            with torch.no_grad():
                stage_output = self.model(stage_data.to(f"cuda:{self.rank}"))

            # 如果不是最后一个阶段，传递给下一个 GPU
            if stage < stages - 1:
                dist.send(stage_output, dst=(self.rank + 1) % self.world_size)
            else:
                return stage_output
```
